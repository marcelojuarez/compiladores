Simbolo inicial P

P -> TR ID() { cuerpo }
cuerpo -> sentencias ret | sentencias

sentencias -> sentencia sentencias | λ
sentencia -> decl | asign | ret 

TR -> void
    | bool
    | int

decl -> cte|var 

var -> tipo id ;
    | tipo asign

cte -> const tipo id = expr ;

asign -> ID = expr ;
      | ID = valBool ;

ret -> return (expr|valBool|λ) ; 

expr -> num
     | ID
     | expr * expr
     | expr / expr
     | expr - expr
     | expr + expr
     
valBool -> true | false | valBool opLog valBool | (valBool) | ID      <-- esta bien?

ID -> letra(letra|numero)*

opLog -> && | || 

numero -> 0|1|2|3|...9

letra -> a|..|z|A|...|Z 

tipo -> bool | int
